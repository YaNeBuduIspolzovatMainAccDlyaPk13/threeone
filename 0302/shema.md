flowchart TD
    %% Блок 1: Основы системы
    A[СИСТЕМА ОЦЕНИВАНИЯ<br>по ФГОС СПО] --> B{ОСНОВА ВСЕГО}
    B --> C[ФГОС + Профстандарты<br>федеральный уровень]
    B --> D[Рабочая программа + ФОС<br>локальный уровень колледжа]

    D --> E[Фонд оценочных средств ФОС]
    E --> E1[Тесты, кейсы,<br>практические задания]
    E --> E2[Билеты,<br>критерии оценки]

    %% Блок 2: Процесс оценки
    E1 & E2 --> F
    subgraph F [ПРОЦЕСС ОЦЕНИВАНИЯ]
        direction LR
        F1[Студент<br>получает задание] --> F2[Выполняет<br>и сдаёт] --> F3[Преподаватель<br>сравнивает с эталоном]
    end

    F3 --> G{Фиксация результата}
    G --> H1[Баллы 0-100]
    G --> H2[Уровень освоения]

    %% Блок 3: Интерпретация
    H1 --> I{Перевод в итоговую оценку}
    H2 --> I

    subgraph I [Итоговая оценка зависит от типа]
        direction LR
        I1[Дисциплина<br>5-балльная шкала]
        I2[Проф. модуль ПМ<br>Освоил / Не освоил]
    end

    I1 --> J[86-100 = 5<br>70-85 = 4<br>51-69 = 3<br>0-50 = 2]
    I2 --> K[Обычно порог 70+ баллов]

    %% Блок 4: Виды контроля
    L[ВИДЫ КОНТРОЛЯ В СЕМЕСТРЕ] --> M
    subgraph M [Этапы по порядку]
        M1[Входной] --> M2[Текущий<br>минимум 3 оценки] --> M3[Рубежный] --> M4[Промежуточная аттестация<br>зачёт/экзамен]
    end

    M4 --> N{Результат}
    N --> O[Сдал →<br>оценка в зачётку]
    N --> P[Не сдал →<br>академич. задолженность]

    P --> Q[Индивидуальный план<br>ликвидации долгов]
    Q --> R{Исход}
    R --> S[Успел → продолжаем]
    R --> T[Не успел → отчисление]

    %% Блок 5: Системные связи
    O & S --> U[Все данные в<br>электронной системе ЭИОС]
    U --> V[Журнал → Портфолио → Отчётность]
    V --> W[Анализ результатов]
    W --> X[Корректировка программ<br>и ФОС на след. год]
    X --> D

**Ключевые принципы (кратко):**
```
1.  СПРАВЕДЛИВОСТЬ     — Правила едины для всех, есть апелляция
2.  ОТКРЫТОСТЬ        — Критерии известны заранее
3.  ПРАКТИКА          — Оценивают реальные проф. умения
4.  СИСТЕМНОСТЬ       — Все этапы связаны: от ФГОС → до оценки в дипломе
```
